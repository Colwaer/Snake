```c++
int GrassReach(link head, link end)
{
    link p = head->next;
    double distance;
    while (p != end)
    {
        distance = fabs(p->x - grassX) * fabs(p->x - grassX) + fabs(p->y - grassY) * fabs(p->y - grassY);
        distance = sqrt(distance);
        if (distance <(double)RADIUS+1 + grassSize) { return 1; }
        p = p->next;
    }
    return 0;
}
```
